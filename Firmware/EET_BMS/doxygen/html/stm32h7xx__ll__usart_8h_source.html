1937  uint32_t OverSampling)
1938 {
1939  uint32_t usartdiv;
1940  uint32_t brrresult = 0x0U;
1941  uint32_t periphclkpresc = (uint32_t)(PeriphClk / (USART_PRESCALER_TAB[(uint8_t)PrescalerValue]));

1944 
1945  if (usartdiv == 0U)
