188 #if defined(CORE_CM7)
189  /* Reset the RCC clock configuration to the default reset state ------------*/
190  /* Increasing the CPU frequency */
191  if(FLASH_LATENCY_DEFAULT > (READ_BIT((FLASH->ACR), FLASH_ACR_LATENCY)))
192  {
193  /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
194  MODIFY_REG(FLASH->ACR, FLASH_ACR_LATENCY, (uint32_t)(FLASH_LATENCY_DEFAULT));

323 void SystemCoreClockUpdate (void)
324 {
325  uint32_t pllp, pllsource, pllm, pllfracen, hsivalue, tmp;
326  uint32_t common_system_clock;
327  float_t fracn1, pllvco;
328 
329  /* Get SYSCLK source -------------------------------------------------------*/
330 
331  switch (RCC->CFGR & RCC_CFGR_SWS)
332  {
333  case RCC_CFGR_SWS_HSI: /* HSI used as system clock source */
334  common_system_clock = (uint32_t) (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3));
335  break;
336 
337  case RCC_CFGR_SWS_CSI: /* CSI used as system clock source */
338  common_system_clock = CSI_VALUE;
339  break;
340 
341  case RCC_CFGR_SWS_HSE: /* HSE used as system clock source */
342  common_system_clock = HSE_VALUE;
343  break;
344 
345  case RCC_CFGR_SWS_PLL1: /* PLL1 used as system clock source */
346 
347  /* PLL_VCO = (HSE_VALUE or HSI_VALUE or CSI_VALUE/ PLLM) * PLLN
348  SYSCLK = PLL_VCO / PLLR
349  */
350  pllsource = (RCC->PLLCKSELR & RCC_PLLCKSELR_PLLSRC);
351  pllm = ((RCC->PLLCKSELR & RCC_PLLCKSELR_DIVM1)>> 4) ;
352  pllfracen = ((RCC->PLLCFGR & RCC_PLLCFGR_PLL1FRACEN)>>RCC_PLLCFGR_PLL1FRACEN_Pos);
353  fracn1 = (float_t)(uint32_t)(pllfracen* ((RCC->PLL1FRACR & RCC_PLL1FRACR_FRACN1)>> 3));
354 
355  if (pllm != 0U)
356  {
357  switch (pllsource)
358  {
359  case RCC_PLLCKSELR_PLLSRC_HSI: /* HSI used as PLL clock source */
360 
361  hsivalue = (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3)) ;
362  pllvco = ( (float_t)hsivalue / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
363 
364  break;
365 
366  case RCC_PLLCKSELR_PLLSRC_CSI: /* CSI used as PLL clock source */
367  pllvco = ((float_t)CSI_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
368  break;
369 
370  case RCC_PLLCKSELR_PLLSRC_HSE: /* HSE used as PLL clock source */
371  pllvco = ((float_t)HSE_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
372  break;
373 
374  default:
375  hsivalue = (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3)) ;
376  pllvco = ((float_t)hsivalue / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
377  break;
378  }
379  pllp = (((RCC->PLL1DIVR & RCC_PLL1DIVR_P1) >>9) + 1U ) ;
380  common_system_clock = (uint32_t)(float_t)(pllvco/(float_t)pllp);
381  }
382  else
383  {
384  common_system_clock = 0U;
385  }
386  break;
387 
388  default:
389  common_system_clock = (uint32_t) (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3));
390  break;
391  }
392 
393  /* Compute SystemClock frequency --------------------------------------------------*/
394  tmp = D1CorePrescTable[(RCC->D1CFGR & RCC_D1CFGR_D1CPRE)>> RCC_D1CFGR_D1CPRE_Pos];
395 
396  /* common_system_clock frequency : CM7 CPU frequency */
397  common_system_clock >>= tmp;
398 
399  /* SystemD2Clock frequency : CM4 CPU, AXI and AHBs Clock frequency */
400  SystemD2Clock = (common_system_clock >> ((D1CorePrescTable[(RCC->D1CFGR & RCC_D1CFGR_HPRE)>> RCC_D1CFGR_HPRE_Pos]) & 0x1FU));
401 
402 #if defined(CORE_CM4)
403  SystemCoreClock = SystemD2Clock;
404 #else
405  SystemCoreClock = common_system_clock;
406 #endif /* CORE_CM4 */
407 }
