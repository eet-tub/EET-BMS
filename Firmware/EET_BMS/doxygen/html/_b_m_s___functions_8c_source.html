64 BmsError_t BMS_CheckLimits()
65 {
66  //Check Under and Overvoltage conditions
67  if(tle_module->overVoltage != 0)
68  {
69  return BMS_CELL_OVERVOLTAGE;
70  }
71 
72  if(tle_module->underVoltage != 0)
73  {
74  return BMS_CELL_UNDERVOLTAGE;
75  }
76 
77  if(tle_module->current > MAXIMUM_CHARGE_CURRENT)
78  {
79  return BMS_OVERCURRENT;
80  }
81 
82  if(tle_module->current < (-MAXIMUM_DISCHARGE_CURRENT))
83  {
84  return BMS_OVERCURRENT;
85  }
86 
87  //Check Cell Voltages
88  for(uint8_t n = 0; n < NUMBEROFCELLS; n++)
89  {
90  float cellvoltage = CONVERT_16_BIT_VALUE_TO_VOLTAGE(tle_module->cellVoltages[n]);
91 
92  if(cellvoltage > MAXIMUM_VOLTAGE)
93  {
94  return BMS_CELL_OVERVOLTAGE;
95  }
96  if(cellvoltage < MINIMUM_VOLTAGE)
97  {
98  return BMS_CELL_UNDERVOLTAGE;
99  }
100  }
101 
102  //Check Temp Sensors
103  for(uint8_t n = 0; n < NUMBEROFTEMPS; n++)
104  {
105  if(tle_module->Temperatures[n] > MAXIMUM_TEMPERATURE)
106  {
107  return BMS_OVERTEMP;
108  }
109  if(tle_module->Temperatures[n] < MINIMUM_TEMPERATURE)
110  {
111  return BMS_UNDERTEMP;
112  }
113  }
114 
115  //Handle General Diagnostic Register Faults
116  GPIO_PinState bms_error_pin = HAL_GPIO_ReadPin(BAT_Relay_GPIO_Port,BAT_Relay_Pin);
117  if(bms_error_pin == GPIO_PIN_SET)
118  {
119  readReg(0, GEN_DIAG, &tle_gendiag_reg); //Read general diagnostic register
120  //TODO Handle General Diagnostic Faults propperly
121  resetGenDiag(); //If all Checks pass, reset Gen Diagnostic interface
122  }
123 
124  return BMS_NO_ERROR;
125 }

146 void BMS_Balancing_Algorithm()
147 {
148 
149  float maximum_voltage = CONVERT_16_BIT_VALUE_TO_VOLTAGE(tle_module->cellVoltages[0]);
150  float minimum_voltage = CONVERT_16_BIT_VALUE_TO_VOLTAGE(tle_module->cellVoltages[0]);
151 
152  float cell_voltages_f[NUMBEROFCELLS];
153 
154  for(uint8_t n = 0; n < NUMBEROFCELLS; n++)
155  cell_voltages_f[n] = CONVERT_16_BIT_VALUE_TO_VOLTAGE(tle_module->cellVoltages[n]);
156  for(uint8_t n = 0; n < NUMBEROFCELLS; n++)
157  {
158  if(cell_voltages_f[n] > maximum_voltage)
159  maximum_voltage = cell_voltages_f[n];
160 
161  if(cell_voltages_f[n] < minimum_voltage)
162  minimum_voltage = cell_voltages_f[n];
163  }
164 
165  if(maximum_voltage < BALANCING_THRESHOLD) //Return if no cell is above balancing threshold
166  {
167  return;
168  }
169 
170  uint16_t channelstobalance = 0; //Balancing Bitmask
171 
172  for(uint8_t n = 0; n < NUMBEROFCELLS; n++) //Check for cells above the Balancing Voltage Delta
173  {
174  if((maximum_voltage-cell_voltages_f[n])<BALANCING_VOLTAGE_DELTA)
175  {
176  channelstobalance |= 1<<n;
177  }
178  }
179 
180  amsConfigBalancing(channelstobalance);
181  amsStartBalancing(BAL_PWM_1000);
182 }
