711 HAL_StatusTypeDef HAL_UARTEx_ReceiveToIdle(UART_HandleTypeDef *huart, uint8_t *pData, uint16_t Size, uint16_t *RxLen,
712  uint32_t Timeout)
713 {
714  uint8_t *pdata8bits;
715  uint16_t *pdata16bits;
716  uint16_t uhMask;
717  uint32_t tickstart;
718 
719  /* Check that a Rx process is not already ongoing */
720  if (huart->RxState == HAL_UART_STATE_READY)
721  {
722  if ((pData == NULL) || (Size == 0U))
723  {
724  return HAL_ERROR;
725  }
726 
727  huart->ErrorCode = HAL_UART_ERROR_NONE;
728  huart->RxState = HAL_UART_STATE_BUSY_RX;
729  huart->ReceptionType = HAL_UART_RECEPTION_TOIDLE;
730  huart->RxEventType = HAL_UART_RXEVENT_TC;
731 
732  /* Init tickstart for timeout management */
733  tickstart = HAL_GetTick();
734 
735  huart->RxXferSize = Size;
736  huart->RxXferCount = Size;
737 
738  /* Computation of UART mask to apply to RDR register */
739  UART_MASK_COMPUTATION(huart);
740  uhMask = huart->Mask;
741 
742  /* In case of 9bits/No Parity transfer, pRxData needs to be handled as a uint16_t pointer */
743  if ((huart->Init.WordLength == UART_WORDLENGTH_9B) && (huart->Init.Parity == UART_PARITY_NONE))
744  {
745  pdata8bits = NULL;
746  pdata16bits = (uint16_t *) pData;
747  }
748  else
749  {
750  pdata8bits = pData;
751  pdata16bits = NULL;
752  }
753 
754  /* Initialize output number of received elements */
755  *RxLen = 0U;
756 
757  /* as long as data have to be received */
758  while (huart->RxXferCount > 0U)
759  {
760  /* Check if IDLE flag is set */
761  if (__HAL_UART_GET_FLAG(huart, UART_FLAG_IDLE))
762  {
763  /* Clear IDLE flag in ISR */
764  __HAL_UART_CLEAR_FLAG(huart, UART_CLEAR_IDLEF);
765 
766  /* If Set, but no data ever received, clear flag without exiting loop */
767  /* If Set, and data has already been received, this means Idle Event is valid : End reception */
768  if (*RxLen > 0U)
769  {
770  huart->RxEventType = HAL_UART_RXEVENT_IDLE;
771  huart->RxState = HAL_UART_STATE_READY;
772 
773  return HAL_OK;
774  }
775  }
776 
777  /* Check if RXNE flag is set */
778  if (__HAL_UART_GET_FLAG(huart, UART_FLAG_RXNE))
779  {
780  if (pdata8bits == NULL)
781  {
782  *pdata16bits = (uint16_t)(huart->Instance->RDR & uhMask);
783  pdata16bits++;
784  }
785  else
786  {
787  *pdata8bits = (uint8_t)(huart->Instance->RDR & (uint8_t)uhMask);
788  pdata8bits++;
789  }
790  /* Increment number of received elements */
791  *RxLen += 1U;
792  huart->RxXferCount--;
793  }
794 
795  /* Check for the Timeout */
796  if (Timeout != HAL_MAX_DELAY)
797  {
798  if (((HAL_GetTick() - tickstart) > Timeout) || (Timeout == 0U))
799  {
800  huart->RxState = HAL_UART_STATE_READY;
801 
802  return HAL_TIMEOUT;
803  }
804  }
805  }
806 
807  /* Set number of received elements in output parameter : RxLen */
808  *RxLen = huart->RxXferSize - huart->RxXferCount;
809  /* At end of Rx process, restore huart->RxState to Ready */
810  huart->RxState = HAL_UART_STATE_READY;
811 
812  return HAL_OK;
813  }
814  else
815  {
816  return HAL_BUSY;
817  }
818 }
